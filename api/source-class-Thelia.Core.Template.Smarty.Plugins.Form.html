 34: /**
 35:  *
 36:  * Plugin for smarty defining blocks and functions for using Form display.
 37:  *
 38:  * blocks :
 39:  *  - {form name="myForm"} ... {/form} => find form named myForm,
 40:  * create an instance and assign this instanciation into smarty variable. Form must be declare into
 41:  * config using <forms> tag
 42:  *
 43:  *  - {form_field form=$form.fieldName} {/form_field} This block find info into the Form field containing by
 44:  * the form paramter. This field must be an instance of FormView. fieldName is the name of your field. This block
 45:  * can output these info :
 46:  *      * $name => name of yout input
 47:  *      * $value => value for your input
 48:  *      * $label => label for your input
 49:  *      * $error => boolean for know if there is error for this field
 50:  *      * $attr => all your attribute for your input (define when you construct programmatically you form)
 51:  *
 52:  *  - {form_error form=$form.fieldName} ... {/form_error} Display this block if there are errors on this field.
 53:  * fieldName is the name of your field
 54:  *
 55:  * Class Form
 56:  * @package Thelia\Core\Template\Smarty\Plugins
 57:  */
 58: class Form extends AbstractSmartyPlugin
 59: {
 60: 
 61:     protected $request;
 62:     protected $parserContext;
 63: 
 64:     protected $formDefinition = array();
 65: 
 66:     public function __construct(Request $request, ParserContext $parserContext)
 67:     {
 68:         $this->request = $request;
 69:         $this->parserContext = $parserContext;
 70:     }
 71: 
 72:     public function setFormDefinition($formDefinition)
 73:     {
 74:         foreach ($formDefinition as $name => $className) {
 75:             if (array_key_exists($name, $this->formDefinition)) {
 76:                 throw new \InvalidArgumentException(sprintf("%s form name already exists for %s class", $name,
 77:                     $className));
 78:             }
 79: 
 80:             $this->formDefinition[$name] = $className;
 81:         }
 82:     }
 83: 
 84:     public function generateForm($params, $content, \Smarty_Internal_Template $template, &$repeat)
 85:     {
 86:         if ($repeat) {
 87: 
 88:             if (empty($params['name'])) {
 89:                 throw new \InvalidArgumentException("Missing 'name' parameter in form arguments");
 90:             }
 91: 
 92:             $instance = $this->createInstance($params['name']);
 93: 
 94:             // Check if parser context contains our form
 95:             $errorForm = $this->parserContext->getErrorForm();
 96: 
 97:             if (null != $errorForm && $errorForm->getName() == $instance->getName()) {
 98: 
 99:                 // Re-use the errored form
100:                 $instance = $errorForm;
101: 
102:                 $this->parserContext->clearErrorForm();
103:             }
104: 
105:             $instance->createView();
106: 
107:             $template->assign("form", $instance);
108: 
109:             $template->assign("form_error", $instance->hasError() ? true : false);
110:             $template->assign("form_error_message", $instance->getErrorMessage());
111:         }
112:         else {
113:             return $content;
114:         }
115:     }
116: 
117:     public function renderFormField($params, $content, \Smarty_Internal_Template $template, &$repeat)
118:     {
119:         if ($repeat) {
120: 
121:             $formFieldView = $this->getFormFieldView($params);
122: 
123:             $template->assign("options", $formFieldView->vars);
124:             $template->assign("name", $formFieldView->vars["full_name"]);
125:             $template->assign("value", $formFieldView->vars["value"]);
126:             $template->assign("label", $formFieldView->vars["label"]);
127: 
128:             $errors = $formFieldView->vars["errors"];
129: 
130:             $template->assign("error", empty($errors) ? false : true);
131: 
132:             if (! empty($errors)) {
133:                 $this->assignFieldErrorVars($template, $errors);
134:             }
135: 
136:             $attr = array();
137: 
138:             foreach ($formFieldView->vars["attr"] as $key => $value) {
139:                 $attr[] = sprintf('%s="%s"', $key, $value);
140:             }
141: 
142:             $template->assign("attr", implode(" ", $attr));
143: 
144:             $formFieldView->setRendered();
145:         }
146:         else {
147:             return $content;
148:         }
149:     }
150: 
151:     public function renderHiddenFormField($params, \Smarty_Internal_Template $template)
152:     {
153:         $field = '<input type="hidden" name="%s" value="%s">';
154: 
155:         $instance = $this->getInstanceFromParams($params);
156: 
157:         $formView = $instance->getView();
158: 
159:         $return = "";
160: 
161:         foreach ($formView->getIterator() as $row) {
162:             if ($this->isHidden($row) && $row->isRendered() === false) {
163:                 $return .= sprintf($field, $row->vars["full_name"], $row->vars["value"]);
164:             }
165:         }
166: 
167:         return $return;
168:     }
169: 
170:     public function formEnctype($params, \Smarty_Internal_Template $template)
171:     {
172:         $instance = $this->getInstanceFromParams($params);
173: 
174:         $formView = $instance->getForm();
175: 
176:         if ($formView->vars["multipart"]) {
177:             return sprintf('%s="%s"',"enctype", "multipart/form-data");
178:         }
179:     }
180: 
181:     public function formError($params, $content, \Smarty_Internal_Template $template, &$repeat)
182:     {
183:         $formFieldView = $this->getFormFieldView($params);
184: 
185:         $errors = $formFieldView->vars["errors"];
186: 
187:         if (empty($errors)) {
188:             return "";
189:         }
190: 
191:         if ($repeat) {
192:             $this->assignFieldErrorVars($template, $errors);
193:         }
194:         else {
195:             return $content;
196:         }
197:     }
198: 
199: 
200:     protected function assignFieldErrorVars(\Smarty_Internal_Template $template, array $errors)
201:     {
202:         $template->assign("message", $errors[0]->getMessage());
203:         $template->assign("parameters", $errors[0]->getMessageParameters());
204:         $template->assign("pluralization", $errors[0]->getMessagePluralization());
205:     }
206: 
207:     protected function isHidden(FormView $formView)
208:     {
209:         return array_search("hidden", $formView->vars["block_prefixes"]);
210:     }
211: 
212:     protected function getFormFieldView($params) {
213:         $instance = $this->getInstanceFromParams($params);
214: 
215:         if (! isset($params['field']))
216:             throw new \InvalidArgumentException("'field' parameter is missing");
217: 
218:         $fieldName = $params['field'];
219: 
220:         if (empty($instance->getView()[$fieldName]))
221:             throw new \InvalidArgumentException(sprintf("Field name '%s' not found in form %s", $fieldName, $instance->getName()));
222: 
223:         return $instance->getView()[$fieldName];
224:     }
225: 
226:     protected function getInstanceFromParams($params) {
227: 
228:         if (empty($params['form'])) {
229:             throw new \InvalidArgumentException("Missing 'form' parameter in form arguments");
230:         }
231: 
232:         $instance = $params["form"];
233: 
234:         if (! $instance instanceof \Thelia\Form\BaseForm) {
235:             throw new \InvalidArgumentException(sprintf("form parameter in form_field block must be an instance of
236:                 \Thelia\Form\BaseForm, instance of %s found", get_class($instance)));
237:         }
238: 
239:         return $instance;
240:     }
241: 
242:     protected function createInstance($name)
243:     {
244:         if (!isset($this->formDefinition[$name])) {
245:             throw new ElementNotFoundException(sprintf("%s form does not exists", $name));
246:         }
247: 
248:         $class = new \ReflectionClass($this->formDefinition[$name]);
249: 
250:         return $class->newInstance(
251:             $this->request,
252:             "form"
253:         );
254:     }
255: 
256:     /**
257:      * @return an array of SmartyPluginDescriptor
258:      */
259:     public function getPluginDescriptors()
260:     {
261:         return array(
262:             new SmartyPluginDescriptor("block", "form", $this, "generateForm"),
263:             new SmartyPluginDescriptor("block", "form_field", $this, "renderFormField"),
264:             new SmartyPluginDescriptor("function", "form_hidden_fields", $this, "renderHiddenFormField"),
265:             new SmartyPluginDescriptor("function", "form_enctype", $this, "formEnctype"),
266:             new SmartyPluginDescriptor("block", "form_error", $this, "formError")
267:         );
268:     }
